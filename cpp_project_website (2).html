#include <iostream>
#include <vector>
using namespace std;

struct student {
  string name;
  string gender;
  int age;
  string department;
  float mark[7];
  string year_of_study;
  float avg;
  string grade;
};
vector<student> students;

string calculate_grade(float avg){
  if(avg ==100)return "+A";
  else if(avg >=90 && avg <100)return "A";
  else if(avg >=85 && avg <90)return "+B";
  else if(avg >=80 && avg <85)return "B";
  else if(avg >=75 && avg <80)return "+C";
  else if(avg >=70 && avg <75)return "C";
  else if(avg >=65 && avg <70)return "+D";
  else if(avg >=60 && avg <65)return "D";
  else if(avg >=55 && avg <60)return "+E";
  else if(avg >=50 && avg <55)return "E";
  else return "F";
}

void add_students(){
  student s;
  cout<<"Enter student name: ";
  cin.ignore();
  getline(cin,s.name);
  cout<<"Enter student age: ";
  cin>>s.age;
  cout<<"Enter student gender: ";
  cin>>s.gender;
  cout<<"Enter student department: ";
  cin>>s.department;
  cout<<"Enter student year of study: ";
  cin.ignore();
  getline(cin,s.year_of_study);

  float total=0;
  cout<<"Enter 7 marks: "<<endl;
  for (int i = 0; i < 7; i++) {
    cout<<"Subject "<<i+1<<": ";
    cin >>s.mark[i];
    total +=s.mark[i];
  }
  s.avg=total/7;
  s.grade=calculate_grade(s.avg);
  students.push_back(s);
  cout<<"\nStudent added"<<endl;
}

void displayall(){
  if (students.empty()){
    cout<<"No record for this student"<<endl;
    return ;
  }
  for(const student& s: students){
    cout<<"\nName: "<<s.name<<endl<<"Average: "<<s.avg<<endl<<"Grade: "<<s.grade<<endl<<"Age: "<<s.age<<endl<<"Gender: "<<s.gender<<endl<<"Department: "<<s.department<<endl<<"Year of study: "<<s.year_of_study<<endl;
  }
}
void search_student(){
  string name;
  cout<<"Enter student name to search: ";
  cin>>name;
  bool found = false;
  for (const student& s: students){
    if (s.name ==name){
      cout<<"\nName: "<<s.name<<endl<<"Average: "<<s.avg<<endl<<"Grade: "<<s.grade<<endl<<"Age: "<<s.age<<endl<<"Gender: "<<s.gender<<endl<<"Department: "<<s.department<<endl<<"Year of study: "<<s.year_of_study<<endl;
      found =true;
      break; 
    }
  }
  if (!found)cout<<"Student not found"<<endl;
}

int main(){
  int choice;
  while(true)
  {
    cout<<"\nStudent grade manager"<<endl;
    cout<<"\n1. Add student\n2. Display all student\n3. Search student\n4. Close \n\nEnter choice: ";
    cin>>choice;
    if(choice==1){
      add_students();}
    else if(choice==2){
      displayall();}
    else if(choice==3){
      search_student();}
    else if (choice ==4){
      cout<<"\nProgram closed succesfully\n"<<endl;
      break;}
    else {
      cout<<"\nInvalid choice"<<endl;
    }
  }
  return 0;
}
